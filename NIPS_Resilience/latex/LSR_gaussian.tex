\section{The two phase boundaries for LSR}

The LSR energy on the $N$-sphere $|\bx|^2 = N$ is
\begin{equation}
H_{\mathrm{LSR}}(\bx) = -\frac{N}{b}\ln\left(\sum_{\mu=1}^{M}\max\!\Big(0,\;1 - b + b\,\frac{\bxi^\mu\!\cdot\!\bx}{N}\Big)\right),
\label{eq:lsr}
\end{equation}
with $M = e^{\alpha N}$ patterns on $S^{N-1}(\!\sqrt{N})$ and sharpness parameter $b > 1$.  The compact support introduces a hard wall at $\varphi_c = (b-1)/b$: pattern~$\mu$ contributes only when $\phi_\mu > \varphi_c$.

In~\citep{icml2026theory}, the LSR retrieval equilibrium overlap $\phi_{\mathrm{LSR}}(T)$ satisfies (Eq.~37 ibid.)
\begin{equation}
(bT+1)\,y^2 - (2+T+Tb)\,y + T = 0\,,\qquad y = 1-\phi\,,
\label{eq:phi_lsr}
\end{equation}
with internal energy density $u_{\mathrm{LSR}}(\phi) = -b^{-1}\ln[1-b(1-\phi)]$ and the kernel-independent geometric entropy $s(\phi) = \tfrac{1}{2}\ln(1-\phi^2)$.  The retrieval free energy density is
\begin{equation}
f_{\mathrm{ret}}(T) = u_{\mathrm{LSR}}(\phi_{\mathrm{LSR}}(T)) - T\,s(\phi_{\mathrm{LSR}}(T))\,.
\label{eq:f_ret_lsr}
\end{equation}

\paragraph{Gaussian approximation.}
The noise floor forms only when the maximum inter-pattern overlap $\varphi_{1,\max}$ exceeds $\varphi_c$.  The Gaussian estimate $\varphi_{1,\max} = \sqrt{2\alpha}$ gives a support threshold and phase boundary (Eqs.~28, 39 of~\citep{icml2026theory}):
\begin{equation}
\alpha_{\mathrm{th}}^{\mathrm{Gauss}} = \frac{\varphi_c^2}{2}\,,\qquad
\alpha_c^{\mathrm{Gauss}}(T) = \frac{1}{2}\big(1-f_{\mathrm{ret}}(T)\big)^2\,.
\label{eq:gauss_lsr}
\end{equation}
For $b = 2+\sqrt{2}$: $\alpha_{\mathrm{th}}^{\mathrm{Gauss}} = 0.250$.  Below $\alpha_{\mathrm{th}}$, no pattern falls within the support and retrieval is perfect at any~$T$.

\paragraph{Exact spherical density.}
Using $\varphi_{1,\max}^{\mathrm{exact}} = \sqrt{1-e^{-2\alpha}}$ from the exact density $f(\phi) = C_N(1-\phi^2)^{(N-3)/2}$:
\begin{equation}
\alpha_{\mathrm{th}}^{\mathrm{exact}} = -\frac{1}{2}\ln(1-\varphi_c^2)\,,\qquad
\alpha_c^{\mathrm{exact}}(T) = -\frac{1}{2}\ln\!\Big(1-\big(1-f_{\mathrm{ret}}(T)\big)^2\Big)\,.
\label{eq:exact_lsr}
\end{equation}
For $b = 2+\sqrt{2}$: $\alpha_{\mathrm{th}}^{\mathrm{exact}} = \tfrac{\ln 2}{2} \approx 0.347$, a 39\% increase.

\paragraph{Comparison.}
Figure~\ref{fig:lsr_phase} shows the two boundaries.  Both have the same qualitative structure (vertical asymptote at $\alpha_{\mathrm{th}}$, curved boundary above it), but the exact density shifts the asymptote from 0.25 to 0.347 and moves the curved part rightward at low~$T$.  The two curves coincide for $T > 1$.

\section{Monte Carlo test}

To distinguish the two predictions, we run the v10 Monte Carlo protocol with the LSR energy~\eqref{eq:lsr}, using $b = 2+\sqrt{2}$.  The v10 parameters (shared with the LSE study): $\alpha \in [0.05, 0.60]$ (step~0.01), 40 temperatures $T \in [0.025, 2.0]$, $M = 10\,000$--$200\,000$, $N_{\mathrm{EQ}} = 2^{14}$, $N_{\mathrm{SAMP}} = 2^{12}$.
